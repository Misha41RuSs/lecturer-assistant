$examDetail = Invoke-RestMethod -Method Get -Uri "http://localhost:8080/exams/f912e23b-1955-4376-97f6-9edc0f059482"

$examDetail.questions[0].options | Format-Table id,text,correct

$qId   = $examDetail.questions[0].id
$optId = ($examDetail.questions[0].options | Where-Object { $_.correct -eq $true } | Select-Object -First 1).id
$qId; $optId
